//CUSTOMER DETAILS
CREATE TABLE customer_details (CUSTOMER_ID number(10) PRIMARY KEY,
FIRST_NAME varchar2(20),
LAST_NAME varchar2(20),
GENDER varchar2(10),
PASSWORD varchar2(20),
MOBILE number(10),
EMAIL varchar2(20) DEFAULT 'NIL',
ADDRESS varchar2(60),
CITY varchar2(15),
STATE varchar2(15),
PINCODE number(6),
CREATED_ON DATE,
STATUS varchar2(10) DEFAULT 'OPEN')

//ACCOUNT DETAILS
CREATE TABLE account_details (ACCOUNT_NO number(15) PRIMARY KEY,
CUSTOMER_ID number(10),
constraint customer_id_constraint FOREIGN KEY(CUSTOMER_ID) REFERENCES customer_details(CUSTOMER_ID))

//TRANSACTION DETAILS
CREATE TABLE transaction_details (TRANSACTION_ID number(10) PRIMARY KEY,
ACCOUNT_NO NUMBER(15),
CREDIT number(15,2) DEFAULT 0,
DEBIT number(15,2) DEFAULT 0,
CUR_BALANCE number(15,2),
TRAN_DATE DATE,
constraint account_no_constraint FOREIGN KEY(ACCOUNT_NO) REFERENCES account_details(ACCOUNT_NO))

//SAVINGS ACCOUNT DETAILS
CREATE TABLE savings_account_details (S_ACCOUNT_NO number(15) PRIMARY KEY,
CUSTOMER_ID number(10),
BALANCE number(15,2) DEFAULT 0,
CREATED_ON DATE,
STATUS varchar2(10) DEFAULT 'OPEN',
LAST_ACCESSED DATE,
CLOSED_ON DATE,
WCOUNTER number(2) DEFAULT 0,
ROI number(5,2) DEFAULT 7.5,
constraint scustomer_id_constraint FOREIGN KEY(CUSTOMER_ID) REFERENCES customer_details(CUSTOMER_ID))

//CURRENT ACCOUNT DETAILS
CREATE TABLE current_account_details (C_ACCOUNT_NO number(15) PRIMARY KEY,
CUSTOMER_ID number(10),
BALANCE number(15,2),
CREATED_ON DATE,
STATUS varchar2(10) DEFAULT 'OPEN',
LAST_ACCESSED DATE,
CLOSED_ON DATE,
constraint ccustomer_id_constraint FOREIGN KEY(CUSTOMER_ID) REFERENCES customer_details(CUSTOMER_ID))

//FD ACCOUNT DETAILS
CREATE TABLE fd_account_details (F_ACCOUNT_NO number(15) PRIMARY KEY,
CUSTOMER_ID number(10),
CREATED_ON DATE,
DURATION number(5),
START_BALANCE number(15,2),
CURR_BALANCE number(15,2),
STATUS varchar2(10) DEFAULT 'OPEN',
CLOSED_ON DATE,
constraint fcustomer_id_constraint FOREIGN KEY(CUSTOMER_ID) REFERENCES customer_details(CUSTOMER_ID))

//LOAN ACCOUNT DETAILS
CREATE TABLE loan_account_details (L_ACCOUNT_NO number(15) PRIMARY KEY,
CUSTOMER_ID number(10),
S_ACCOUNT_NO number(15),
LOAN_AMOUNT number(15,2),
REPAY_TERM number(15,2),
constraint lcustomer_id_constraint FOREIGN KEY(CUSTOMER_ID) REFERENCES customer_details(CUSTOMER_ID),
constraint s_account_constraint FOREIGN KEY(S_ACCOUNT_NO) REFERENCES savings_account_details(S_ACCOUNT_NO))

//USER DETAILS
CREATE TABLE users (USER_NAME varchar2(30) PRIMARY KEY, PASSWORD varchar2(20), STATUS varchar2(10) DEFAULT 'OPEN')
